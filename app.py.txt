from flask import Flask, request, jsonify
from flask_cors import CORS

app = Flask(__name__)
CORS(app)  # Enable CORS so the frontend can call the backend

@app.route('/chat', methods=['POST'])
def chat():
    data = request.json
    user_message = data.get("message", "")
    reply = f"You said: {user_message}"  # Replace this with your AI logic
    return jsonify({"reply": reply})

if __name__ == '__main__':
    app.run(debug=True)
